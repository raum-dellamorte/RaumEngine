#version 140

in vec2 pass_textureCoords;

out vec4 out_colour;

uniform vec3 colour;
uniform sampler2D fontAtlas;

const float width = 0.5;
const float edge = 0.1;
const float widthBorder = 0.6;
const float edgeBorder = 0.1;

const vec2 offset = vec2(0.001, 0.001);

const vec3 outlineColour = vec3(1.0, 0.0, 0.0);

void main(void){
  
  float distance = 1.0 - texture(fontAtlas, pass_textureCoords).a;
  float alpha = 1.0 - smoothstep(width, width + edge, distance);
  float distanceOffset = 1.0 - texture(fontAtlas, pass_textureCoords + offset).a;
  float alphaBorder = 1.0 - smoothstep(widthBorder, widthBorder + edgeBorder, distanceOffset);
  
  float alphaMaster = alpha + (1.0 - alpha) * alphaBorder;
  vec3 colourMaster = mix(outlineColour, colour, alpha / alphaMaster);
  
  out_colour = vec4(colourMaster, alphaMaster);
  
}
